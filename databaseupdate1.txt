-- Policy: cho phép người nhận đánh dấu message của họ là đã đọc
CREATE POLICY allow_receiver_update_is_read
ON public.chat_messages
FOR UPDATE
USING ( receiver_id = current_setting('request.jwt.claim.user_id')::int )
WITH CHECK ( receiver_id = current_setting('request.jwt.claim.user_id')::int AND is_read = true );


-- RPC: mark conversation read (SECURITY DEFINER)
create or replace function public.mark_conversation_read(p_conv_id integer)
returns void
language plpgsql
security definer
as $$
begin
  -- Bảo đảm chỉ cập nhật những message thuộc conversation và receiver = user id từ JWT claim
  update public.chat_messages
  set is_read = true
  where conversation_id = p_conv_id
    and receiver_id = current_setting('request.jwt.claim.user_id')::int;
end;
$$;


-- RPC: mark_conversation_read
-- SECURITY DEFINER function to safely mark all messages in a conversation as read
-- for the calling user. Call via: SELECT public.mark_conversation_read(<conv_id>);
-- or from client: await sb.rpc('mark_conversation_read', { p_conv_id: convId });
CREATE OR REPLACE FUNCTION public.mark_conversation_read(p_conv_id integer)
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
  -- update only messages that belong to the conversation and where the receiver
  -- matches the user_id in the JWT claim to avoid privilege escalation
  UPDATE public.chat_messages
  SET is_read = true
  WHERE conversation_id = p_conv_id
    AND receiver_id = current_setting('request.jwt.claim.user_id')::int;
END;
$$;

GRANT EXECUTE ON FUNCTION public.mark_conversation_read(integer) TO public;

-- WHEEL / VOUCHER FEATURE TABLES & RPCs
-- Run these in Supabase SQL editor. These functions use SECURITY DEFINER
-- and are pragmatic for class projects where Supabase Auth is not used.
-- Review and adjust limits (spin per day) as needed.

-- Wheel segments (admin-managed)
create table if not exists wheel_segments (
  segment_id int generated by default as identity primary key,
  label varchar(150) not null,
  discount_percent int default 0 check (discount_percent >= 0 and discount_percent <= 100),
  min_order_amount numeric(10,2) default 0 check (min_order_amount >= 0),
  probability numeric(10,6) not null default 0,
  active boolean default true,
  created_at timestamptz default now()
);

-- Voucher master
create table if not exists vouchers (
  voucher_id int generated by default as identity primary key,
  voucher_code varchar(50) not null unique,
  discount_percent int not null check (discount_percent >= 0 and discount_percent <= 100),
  min_order_amount numeric(10,2) default 0 check (min_order_amount >= 0),
  expires_at timestamptz,
  created_at timestamptz default now(),
  source_segment int references wheel_segments(segment_id)
);

-- User assigned vouchers
create table if not exists user_vouchers (
  user_voucher_id int generated by default as identity primary key,
  user_id int not null references users(user_id) on delete cascade,
  voucher_id int not null references vouchers(voucher_id) on delete cascade,
  is_used boolean default false,
  assigned_at timestamptz default now()
);

-- Track user spins for rate limiting / analytics
create table if not exists user_spins (
  spin_id int generated by default as identity primary key,
  user_id int not null references users(user_id) on delete cascade,
  segment_id int references wheel_segments(segment_id),
  created_at timestamptz default now()
);

-- SECURITY DEFINER RPC: create_voucher_for_user
-- Usage: SELECT public.create_voucher_for_user(p_user_id := 3, p_segment_id := 1);
-- Returns the created user_vouchers row and voucher info as JSON.
CREATE OR REPLACE FUNCTION public.create_voucher_for_user(p_user_id integer, p_segment_id integer)
RETURNS jsonb
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  seg row;
  user_spin_count int;
  v_id int;
  uv_id int;
  v_code text;
  result jsonb;
BEGIN
  -- basic validations
  SELECT * INTO seg FROM wheel_segments WHERE segment_id = p_segment_id FOR UPDATE;
  IF NOT FOUND OR NOT seg.active THEN
    RAISE EXCEPTION 'Invalid segment';
  END IF;

  -- rate limit: max 5 spins per 24 hours (adjust as needed)
  SELECT count(*) INTO user_spin_count FROM user_spins WHERE user_id = p_user_id AND created_at > now() - interval '24 hours';
  IF user_spin_count >= 5 THEN
    RAISE EXCEPTION 'Spin limit reached for 24h';
  END IF;

  -- record spin
  INSERT INTO user_spins (user_id, segment_id) VALUES (p_user_id, p_segment_id) RETURNING spin_id INTO STRICT uv_id;

  -- create voucher master row
  v_code := 'WHEEL' || substring(md5(random()::text || clock_timestamp()::text),1,8);
  INSERT INTO vouchers (voucher_code, discount_percent, min_order_amount, expires_at, source_segment)
    VALUES (v_code, seg.discount_percent, seg.min_order_amount, now() + interval '30 days', seg.segment_id)
    RETURNING voucher_id INTO v_id;

  -- assign to user
  INSERT INTO user_vouchers (user_id, voucher_id) VALUES (p_user_id, v_id) RETURNING user_voucher_id INTO uv_id;

  result := jsonb_build_object('user_voucher_id', uv_id, 'voucher_id', v_id, 'voucher_code', v_code, 'discount_percent', seg.discount_percent, 'min_order_amount', seg.min_order_amount);
  RETURN result;
END;
$$;

GRANT EXECUTE ON FUNCTION public.create_voucher_for_user(integer, integer) TO public;

-- Pragmatic RPC to mark conversation read when no JWT auth is used (class project)
-- Usage: SELECT public.mark_conversation_read_for_user(p_conv_id := 1, p_user_id := 3);
CREATE OR REPLACE FUNCTION public.mark_conversation_read_for_user(p_conv_id integer, p_user_id integer)
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
  UPDATE public.chat_messages
  SET is_read = true
  WHERE conversation_id = p_conv_id
    AND receiver_id = p_user_id;
END;
$$;
GRANT EXECUTE ON FUNCTION public.mark_conversation_read_for_user(integer, integer) TO public;
